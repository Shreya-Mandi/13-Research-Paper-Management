1. "http://localhost:3002/Login/"
    method: POST

    request:
    {
        id : String,
        pwd : String
    }

    response:
    {
        invalidRequest : true / false,
        status : true / false,
        errMsg : String,

        valid : true / false,
        type : "student" / "faculty",
        name : String
    }


2. "http://localhost:3002/Register/"
    method: POST

    request:
    {
        id : String,
        pwd : String,
        type : "student" / "faculty",

        (if type == "student")
        details: {
                     firstName : String,
                     lastName : String,
                     dept : "CSE" / "ECE" / "EEE" / "AIML" / "ME",
                     sem : Int,
                     sec : Char
                 }

        (if type == "faculty")
        details: {
                     firstName : String,
                     lastName : String,
                     dept : "CSE" / "ECE" / "EEE" / "AIML" / "ME"
                     domain : String
                 }
    }

    response:
    {
        invalidRequest : true / false,
        status : true / false,
        errMsg : String
    }

                                     ]
                       }, ...
                   ]
    }
